pipeline {
    agent any

    stages {
        stage('Build and Test') {
            steps {
                // Clean and run tests, also generate HTML report
                sh 'mvn clean verify'
            }
        }
    }

    post {
        always {
            // Publish test result reports (JUnit XML)
            junit '**/target/surefire-reports/*.xml'

            // Publish pretty HTML report
            publishHTML([
                reportDir: 'target/site',
                reportFiles: 'surefire-report.html',
                reportName: 'Surefire HTML Report',
                keepAll: true,
                allowMissing: false,
                alwaysLinkToLastBuild: true
            ])
        }
    }
}
