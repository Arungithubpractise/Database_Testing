pipeline {
    agent any

    tools {
        maven 'Maven 3.8.6' // Change to your configured Maven version in Jenkins
        jdk 'JDK 11'        // Or whatever version you're using
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build') {
            steps {
                sh 'mvn clean install'
            }
        }

        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
    }

    post {
        always {
            // Publish test results
            junit 'target/surefire-reports/*.xml'

            // (Optional) Publish HTML reports if you have them
            // Make sure to generate an HTML report via plugin like Maven Surefire Report Plugin
            publishHTML(target: [
                reportName : 'JUnit Report',
                reportDir  : 'target/site/surefire-report',
                reportFiles: 'index.html',
                keepAll    : true,
                alwaysLinkToLastBuild: true
            ])
        }
    }
}
